#/bin/sh

for m in 5D 4D 3D 2D 1D; do

#logical
sed 's/IntOrReal/LOGICAL/g' < 'DataVolume5DClassCreateAndDestroyTest'$m'Template.f90' > 'DataVolume5DClassCreateAndDestroyTest'$m'LogicalTemplate.f90'

#real
sed 's/IntOrReal/REAL(KIND=VarTypeDef)/g' < 'DataVolume5DClassCreateAndDestroyTest'$m'Template.f90' > 'DataVolume5DClassCreateAndDestroyTest'$m'RealTemplate.f90'

#integer
sed 's/IntOrReal/INTEGER(KIND=VarTypeDef)/g' < 'DataVolume5DClassCreateAndDestroyTest'$m'Template.f90' > 'DataVolume5DClassCreateAndDestroyTest'$m'IntTemplate.f90'

for n in 4 8; do
 sed 's/VarTypeDef/int'$n'Def/g' < 'DataVolume5DClassCreateAndDestroyTest'$m'IntTemplate.f90' > scratch.txt
 sed 's/varType/int'$n'/g' < scratch.txt > scratch2.txt
 sed 's/VarType/Int'$n'/g' < scratch2.txt > 'DataVolume5DClassCreateAndDestroyTest'$m'Int'$n'.f90'
done

for n in 4 8; do
 sed 's/VarTypeDef/real'$n'Def/g' < 'DataVolume5DClassCreateAndDestroyTest'$m'RealTemplate.f90' > scratch.txt
 sed 's/varType/real'$n'/g' < scratch.txt > scratch2.txt
 sed 's/VarType/Real'$n'/g' < scratch2.txt > 'DataVolume5DClassCreateAndDestroyTest'$m'Real'$n'.f90'
done

# logical
 sed 's/varType/logical/g' < 'DataVolume5DClassCreateAndDestroyTest'$m'LogicalTemplate.f90' > scratch.txt
 sed 's/VarType/Logical/g' < scratch.txt > 'DataVolume5DClassCreateAndDestroyTest'$m'Logical.f90'

# cleanup

rm scratch*.txt
rm 'DataVolume5DClassCreateAndDestroyTest'$m'RealTemplate.f90'
rm 'DataVolume5DClassCreateAndDestroyTest'$m'IntTemplate.f90'
rm 'DataVolume5DClassCreateAndDestroyTest'$m'LogicalTemplate.f90'

done
