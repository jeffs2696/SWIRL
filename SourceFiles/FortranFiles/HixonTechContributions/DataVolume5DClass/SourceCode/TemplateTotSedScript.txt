#/bin/sh

#logical
sed 's/IntOrReal/LOGICAL/' < DataVolume5DClassTotRoutinesTemplate.f90 > DataVolume5DClassTotRoutinesLogicalTemplate.f90

#real
sed 's/IntOrReal/REAL(KIND=VarTypeDef)/' < DataVolume5DClassTotRoutinesTemplate.f90 > DataVolume5DClassTotRoutinesRealTemplate.f90

#integer
sed 's/IntOrReal/INTEGER(KIND=VarTypeDef)/' < DataVolume5DClassTotRoutinesTemplate.f90 > DataVolume5DClassTotRoutinesIntTemplate.f90

for n in 4 8; do
 sed 's/VarTypeDef/int'$n'Def/' < DataVolume5DClassTotRoutinesIntTemplate.f90 > scratch.txt
 sed 's/varType/int'$n'/' < scratch.txt > scratch2.txt
 sed 's/VarType/Int'$n'/' < scratch2.txt > 'DataVolume5DClassTotInt'$n'Routines.f90'
done

for n in SP DP; do
 sed 's/VarTypeDef/real'$n'Def/' < DataVolume5DClassTotRoutinesRealTemplate.f90 > scratch.txt
 sed 's/varType/real'$n'/' < scratch.txt > scratch2.txt
 sed 's/VarType/Real'$n'/' < scratch2.txt > 'DataVolume5DClassTotReal'$n'Routines.f90'
done

# logical
 sed 's/varType/logical/' < DataVolume5DClassTotRoutinesLogicalTemplate.f90 > scratch.txt
 sed 's/VarType/Logical/' < scratch.txt > DataVolume5DClassTotLogicalRoutines.f90

# cleanup

rm scratch*.txt
rm DataVolume5DClassTotRoutinesRealTemplate.f90
rm DataVolume5DClassTotRoutinesIntTemplate.f90
rm DataVolume5DClassTotRoutinesLogicalTemplate.f90
